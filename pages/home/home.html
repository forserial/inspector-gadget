Inspector Gadget

Inspector Gadget is an AngularJS popover implementation that lets you create rich popover content with HTML and all of angular's goodies. This goes beyond angular bootstrap's popover, which gives a much simpler popover interface.

For usage, installation, and what-not, check out the github repo. Enough yaking, here's a few examples:

Hover to see company sales! Latest Prices

Markup

Controller

<inspector-gadget data-container="body">
  <a class="btn">Hover to check computer prices</a>

  <inspector-title style="width: 100%">
    Latest Prices
  </inspector-title>

  <inspector-content>
    <div ng-controller="ChartsCtrl">
      <div 
        data-ac-chart="'bar'" 
        data-ac-data="data" 
        data-ac-config="config" 
        class="chart">
      </div>
    </div>
  </inspector-content>
</inspector-gadget>

  

angular.module('myApp').controller('ChartsCtrl', function($scope) {
  $scope.config = {
    title: 'Products',
    tooltips: true,
    labels: false,
    mouseover: function() {},
    mouseout: function() {},
    click: function() {},
    legend: {
      display: true,
      //could be 'left, right'
      position: 'right'
    }
  };

  $scope.data = {
    series: ['Sales', 'Income', 'Expense', 'Laptops', 'Keyboards'],
    data: [{
      x: "Laptops",
      y: [100, 500, 0],
      tooltip: "this is tooltip"
    }, {
      x: "Desktops",
      y: [300, 100, 100]
    }, {
      x: "Mobiles",
      y: [351]
    }, {
      x: "Tablets",
      y: [54, 0, 879]
    }]
  };
});

Simpler example, free running counter, dynamically changing popover

Here's another example, using a simple free running counter:

Mouse me! {{tenSecTimer}}

Marry had a little lamb its fleece was white as snow

Hello World! Its {{tenSecTimer * 10}} o'clock Ipusum lorem est {{timer}}

Markup

Controller

<div ng-controller='HomeCtrl'>  
    <inspector-gadget data-placement="bottom">
        <p>Mouse me! {{tenSecTimer}}</p>
          <p>Marry had a little lamb its fleece was white as snow</p>
      <inspector-title>
        Hello <b>World! Its {{tenSecTimer * 10}} o'clock</b>
      </inspector-title>
      <inspector-content>
        Ipusum lorem <i>est</i> {{timer}}
      </inspector-content>
    </inspector-gadget>
</div>
  
angular.module('myApp').controller('HomeCtrl', function($scope, $interval) {
  $scope.timer = 0;
  $scope.tenSecTimer = 0;

  $interval(function tick() {
    $scope.timer++;
    if ($scope.timer % 10 === 0) {
      $scope.tenSecTimer++;
    }
  }, 1000);
});
